		Enter the no. <br>
		<input type="number" name="val" id="num"> <br>
		<input type="button" name="check" onclick="check1()" value="Check">
		Result is <span id="res"></span>....!
		</form>

	<script>
		function check1(){
			var x = document.getElementById("num").value;
			
			if(x==""){
				document.getElementById("res").innerHTML =  "Enter  value";
			}
			else if(x<0)
				document.getElementById("res").innerHTML =  "Enter positive value";
			else{
			if(x%3==0 && x%7==0)
				document.getElementById("res").innerHTML =  "Divisible by 3 and 7";
			else if(x%3==0 && x%7!=0)
				document.getElementById("res").innerHTML =  "Divisible by only 3 and not 7";
			else if(x%3!=0 && x%7==0)
				document.getElementById("res").innerHTML =  "Divisible by only 7 and not 3";
			else
				document.getElementById("res").innerHTML =  "Not divisible by both 3 or 7";
		}
		}
	</script>
